main.py

import retro

def main():
    env = retro.make(game='Airstriker-Genesis')
    #env = retro.make(game='AlteredBeast-Genesis')
    obs = env.reset()
    while True:
        ac = env.action_space.sample()
        ob, rew, terminated, truncated, info = env.step(ac)
        env.render()
        if terminated:
            obs = env.reset()
    env.close()

if __name__ == "__main__":
    main()
