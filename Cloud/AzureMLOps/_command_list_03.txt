01. 	az ml online-endpoint
export RESOURCE_GRP="SteveProGroup"
export ML_WORKSPACE="SteveProWorkspace"
export MYENDPOINT="online-endpoint"

az ml online-endpoint list -g $RESOURCE_GRP -w $ML_WORKSPACE -o table
az ml online-endpoint show -g $RESOURCE_GRP -w $ML_WORKSPACE -n $MYENDPOINT
az ml online-endpoint create --file endpoint.yaml -g $RESOURCE_GRP -w $ML_WORKSPACE
az ml online-endpoint update --file endpoint.yaml -g $RESOURCE_GRP -w $ML_WORKSPACE
az ml online-endpoint delete -g $RESOURCE_GRP -w $ML_WORKSPACE -n $MYENDPOINT

Long-form
-g	--resource-group
-w	--workspace-name
az ml online-endpoint list --resource-group $RESOURCE_GRP --workspace-name $ML_WORKSPACE --output table
az ml online-endpoint show --resource-group $RESOURCE_GRP --workspace-name $ML_WORKSPACE --name $MYENDPOINT
az ml online-endpoint create --file endpoint.yaml --resource-group $RESOURCE_GRP --workspace-name $ML_WORKSPACE
az ml online-endpoint update --file endpoint.yaml --resource-group $RESOURCE_GRP --workspace-name $ML_WORKSPACE
az ml online-endpoint delete --resource-group $RESOURCE_GRP --workspace-name $ML_WORKSPACE --name $MYENDPOINT
