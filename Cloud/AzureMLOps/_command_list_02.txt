export RESOURCE_GRP="SteveProGroup"
export ML_WORKSPACE="SteveProWorkspace"
export ENVIRON="test-model"
export VERSION="1"

az ml environment list -g $RESOURCE_GRP -w $ML_WORKSPACE -o table
az ml environment list -g $RESOURCE_GRP -w $ML_WORKSPACE -n $ENVIRON
# verbose
az ml environment list -g $RESOURCE_GRP -w $ML_WORKSPACE -n $ENVIRON -o table
# short key/val - all versions for model
az ml environment list -g $RESOURCE_GRP -w $ML_WORKSPACE -n $ENVIRON -o table --query "[].{Name:name, Version:version}"
az ml environment show -g $RESOURCE_GRP -w $ML_WORKSPACE -n $ENVIRON -v $VERSION
az ml environment create --file model.yaml -g $RESOURCE_GRP -w $ML_WORKSPACE
az ml environment update -g $RESOURCE_GRP -w $ML_WORKSPACE -n $ENVIRON -v $VERSION
az ml environment archive -g $RESOURCE_GRP -w $ML_WORKSPACE -n $ENVIRON -v $VERSION

# Loop archive multiple environments
for v in 1 2 3 4 5 6 7; do
  az ml environment archive -g $RESOURCE_GRP -w $ML_WORKSPACE -n $ENVIRON -v $v
done
