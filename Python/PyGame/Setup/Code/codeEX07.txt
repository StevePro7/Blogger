&nbsp;&nbsp;import pygame
&nbsp;&nbsp;pygame.init()
&nbsp;&nbsp;pygame.mixer.init()
&nbsp;&nbsp;
&nbsp;&nbsp;screen = pygame.display.set_mode((640, 480))
&nbsp;&nbsp;pygame.display.set_caption(&quot;Hello Music&quot;)
&nbsp;&nbsp;
&nbsp;&nbsp;font = pygame.font.Font(None, 48)  # Built-in default font
&nbsp;&nbsp;pygame.mixer.music.load(&quot;music.mp3&quot;)
&nbsp;&nbsp;
&nbsp;&nbsp;def render_text(text, color=(255, 255, 255)):
&nbsp;&nbsp;    return font.render(text, False, color)
&nbsp;&nbsp;
&nbsp;&nbsp;text1 = render_text(&quot;Hello Music!&quot; )
&nbsp;&nbsp;text2 = render_text(&quot;Press Up - play music&quot;)
&nbsp;&nbsp;text3 = render_text(&quot;Press Down - stop music&quot;)
&nbsp;&nbsp;
&nbsp;&nbsp;running = True
&nbsp;&nbsp;while running:
&nbsp;&nbsp;    for event in pygame.event.get():
&nbsp;&nbsp;        if event.type == pygame.QUIT:
&nbsp;&nbsp;            running = False
&nbsp;&nbsp;
&nbsp;&nbsp;    screen.fill((0, 0, 0))
&nbsp;&nbsp;    screen.blit(text1, (220, 140))
&nbsp;&nbsp;    screen.blit(text2, (160, 180))
&nbsp;&nbsp;    screen.blit(text3, (160, 220))
&nbsp;&nbsp;
&nbsp;&nbsp;    keys = pygame.key.get_pressed()
&nbsp;&nbsp;    if keys[pygame.K_UP]:
&nbsp;&nbsp;        pygame.mixer.music.play()
&nbsp;&nbsp;    if keys[pygame.K_DOWN]:
&nbsp;&nbsp;        pygame.mixer.music.stop()
&nbsp;&nbsp;
&nbsp;&nbsp;    pygame.display.flip()
&nbsp;&nbsp;
&nbsp;&nbsp;pygame.quit()
